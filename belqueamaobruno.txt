<script>
  const btnYes = document.getElementById('btnYes');
  const btnNo = document.getElementById('btnNo');
  const message = document.getElementById('message');
  const finalMessage = document.getElementById('finalMessage');
  const title = document.querySelector('h1');

  let noClickCount = 0;

  btnYes.addEventListener('click', () => {
    if(btnYes.textContent === 'Sim'){
      btnYes.textContent = 'Você tem certeza? me ama mesmo? sim?';
      title.textContent = 'Quer namorar comigo, meu amor?';
      message.textContent = '';
    } else if(btnYes.textContent === 'Você tem certeza? me ama mesmo? sim?'){
      title.textContent = 'Você quer mesmo ser meu neném pra sempre? Sim??';
      message.textContent = '';
    } else {
      btnYes.style.display = 'none';
      btnNo.style.display = 'none';
      finalMessage.style.display = 'block';
    }
  });

  const noMessages = [
    "Amor, porque não? ",
    "Neném? ",
    "Você não me ama mais? ",
    "Por favor, amor...",
    "Diz que sim, vida... Eu imploro meu neném"
  ];

  btnNo.addEventListener('click', () => {
    if(noClickCount < noMessages.length){
      btnNo.textContent = noMessages[noClickCount];
      btnYes.style.transform = `scale(${1 + noClickCount * 0.3})`;
      noClickCount++;
    } 

    if(noClickCount === noMessages.length){
      btnNo.classList.add('hidden');
      btnYes.classList.add('fullscreen');
    }
  });
</script>
